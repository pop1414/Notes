  usecase "预定房间" as UC_Reserve

  usecase "计算总费用\n(季节房价+工作日折扣)" as UC_Fee
  usecase "收取10%定金" as UC_Deposit

  usecase "取消预定" as UC_Cancel
  usecase "退还定金" as UC_Refund

  usecase "维护房价和\n季节/折扣规则" as UC_Price
  usecase "生成并打印\n一周预定清单" as UC_Report
}

Customer --> UC_Reserve
Customer --> UC_Cancel

Clerk --> UC_Price

' —— 用例之间的包含与扩展关系 ——
UC_Reserve .> UC_Record  : <<include>>
UC_Reserve .> UC_Fee     : <<include>>
UC_Reserve .> UC_Deposit : <<include>>

  - 周一至周五：房费半价

end note
