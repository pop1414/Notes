@startuml
interface Beverage{
	+cost(): double
}

class Coffee{
	+cost(): double
}

class Tea{
	+cost(): double
}

abstract class CondimentDecorator{
	-beverage: Beverage
	+cost(): double
}

class Milk{
	+cost(): double
}

class Sugar{
	+cost(): double
}

Coffee ..|> Beverage: implements
Tea ..|> Beverage: implements
CondimentDecorator ..|> Beverage: implements
CondimentDecorator o--> Beverage: Aggregation
Milk --|> CondimentDecorator: extends
Sugar --|> CondimentDecorator: extends


note bottom of CondimentDecorator
	double cost(){
		return beverage.cost();
	}
end note

note bottom of Milk
	double cost(){
		return super.cost() + 0.5; 
	}
end note

@enduml
