class Client{
}

' Target
interface Robot{
    +cry(): void
    +move(): void
}

' Adaptee
class Dog{
	+wang(): void
	+run(): void
}

' Adapter
class DogAdapter{
	+cry(): void
	+move(): void
	-dog: Dog
}

DogAdapter ..|> Robot: implements
DogAdapter *--> Dog: composition
Client --> Robot: Association

	cry() { 
		dog.wang(); 
	} 
	move() { 
		dog.run();
