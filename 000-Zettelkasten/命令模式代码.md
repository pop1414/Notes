@startuml
interface Command{
	+execute(): void
}

class AddCommand{
	-calculatorCore: Calculator
	+execute(): void
}

class SubtractCommand{
	-calculatorCore: Calculator
	+execute(): void
}

class Calculator{
	-value: double
	+add(): void
	+subtract():void
	+getValue(): double
}

class CalculatorInvoker{
	-commandQueue: CommandQueue	
	+compute(): void
}

class CommandQueue{
	-commands: List<Command>
	+addCommand(): void
	+removeCommand():void
	+execute(): void	
}

AddCommand ..|> Command: implements
SubtractCommand ..|> Command: implements
AddCommand --> Calculator: Association
SubtractCommand --> Calculator: Association
CalculatorInvoker o--> CommandQueue: Aggregation
CommandQueue o--> Command: Aggregation

note bottom of CalculatorInvoker
	in compute():
		commandQueue.execute();
end note

note bottom of CommandQueue
	in execute():
		for(Command command : commands){
			command.execute();
		}
end note

note bottom of AddCommand
	in execute():
		this.calculatorCore.add();
end note

@enduml
